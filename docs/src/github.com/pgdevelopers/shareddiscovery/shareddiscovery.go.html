    60  // GetConfig uses the provided `APIToken` to get the correct
    61  // configuration from the specified `tableName`.
    62  func (service SharedDiscovery) GetConfig(ctx context.Context, apiToken string, query QueryInput) (map[string]interface{}, error) {
    63  	_, configSpan := beeline.StartSpan(ctx, "GetConfig")
    64  	configSpan.AddField("workspace", query.Workspace)
    65  
    66  	// dynamically build attribute values
    67  	searchAttributes := map[string]*dynamodb.AttributeValue{
    68  		"apiToken": {
    69  			S: &apiToken,
    70  		},
    71  	}
    72  	searchAttributes = addNeededSearchAttributes(searchAttributes, query)
    73  
    74  	appResult, err := service.DynamodbSvc.GetItem(&dynamodb.GetItemInput{
    75  		TableName: &query.Workspace,
    76  		Key:       searchAttributes,
    77  	})
    78  
    79  	if err != nil {
    80  		return nil, err
    81  	}
    82  
    83  	var discovery map[string]interface{}
    84  	err = dynamodbattribute.UnmarshalMap(appResult.Item, &discovery)
    85  	if err != nil {
    86  		return nil, err
    87  	}
    88  
    89  	configSpan.Send()
    90  	return discovery, nil
    91  }

   101  // AdminGetAPIToken queries the dynamo table using the provided query
   102  // and returns the correct apiToken for the caller to use to then make
   103  // a request using the GetConfig call.
   104  // It first validates the HMAC signature against the provided secretKey/query params
   105  // to verify the caller is who they say they are.
   106  func (service SharedDiscovery) AdminGetAPIToken(ctx context.Context, secretKey string, query QueryInput) (string, error) {
   107  	_, getAPIKeySpan := beeline.StartSpan(ctx, "adminGetAPIToken")
   108  	defer getAPIKeySpan.Send()
   109  
   110  	// validate signature
   111  	if !validateSignature(ctx, query, secretKey) {
   112  		getAPIKeySpan.AddField("error.message", "invalid signature detected")
   113  		return "", errors.New("invalid signature")
   114  	}
   115  
   116  	// run query
   117  	items, err := getAPITokenQuery(ctx, service, query)
   118  	if err != nil {
   119  		getAPIKeySpan.AddField("error.message", err.Error())
   120  		return "", err
   121  	}
   122  
   123  	// parse token
   124  	return parseAPIToken(ctx, items)
   125  }

   127  func validateSignature(ctx context.Context, query QueryInput, secretKey string) bool {
   128  	_, validateSignatureSpan := beeline.StartSpan(ctx, "validateSignature")
   129  	validateSignatureSpan.AddField("query.object", query)
   130  	message := messageFromQuery(query)
   131  	validateSignatureSpan.AddField("query.string", message)
   132  	mac := hmac.New(sha256.New, []byte(secretKey))
   133  	mac.Write([]byte(message))
   134  	expectedMAC := mac.Sum(nil)
   135  
   136  	decoded, err := hex.DecodeString(query.Signature)
   137  	if err != nil {
   138  		validateSignatureSpan.AddField("error.message", fmt.Sprintf("unable to decode signature: %s", err.Error()))
   139  		return false
   140  	}
   141  
   142  	return hmac.Equal([]byte(decoded), expectedMAC)
   143  }

   145  func messageFromQuery(query QueryInput) string {
   146  	// order the query string keys alphabetically
   147  	keys := make([]string, len(query.QueryString))
   148  
   149  	i := 0
   150  	for k := range query.QueryString {
   151  		keys[i] = k
   152  		i++
   153  	}
   154  	sort.Strings(keys)
   155  
   156  	var message = ""
   157  	for _, val := range keys {
   158  		message = fmt.Sprintf("%s%s", message, query.QueryString[val])
   159  	}
   160  	return message
   161  }

   163  func getAPITokenQuery(ctx context.Context, service SharedDiscovery, query QueryInput) ([]map[string]*dynamodb.AttributeValue, error) {
   164  	_, getAPIKeySpan := beeline.StartSpan(ctx, "getAPITokenQuery")
   165  	defer getAPIKeySpan.Send()
   166  	if query.AppName == "" {
   167  		appResult, err := service.DynamodbSvc.Scan(&dynamodb.ScanInput{
   168  			TableName:        &query.Workspace,
   169  			FilterExpression: aws.String("environment = :e and countryCode = :c and brandName = :b"),
   170  			ExpressionAttributeValues: map[string]*dynamodb.AttributeValue{
   171  				":e": {
   172  					S: aws.String(query.Environment),
   173  				},
   174  				":c": {
   175  					S: aws.String(query.Country),
   176  				},
   177  				":b": {
   178  					S: aws.String(query.Brand),
   179  				},
   180  			},
   181  		})
   182  		if err != nil {
   183  			getAPIKeySpan.AddField("error.message", fmt.Sprintf("Unable to get apiToken from discovery v3 admin: %s", err.Error()))
   184  			getAPIKeySpan.AddField("query.values", fmt.Sprintf("%s,%s,%s", query.AppName, query.Country, query.Environment))
   185  			return nil, err
   186  		}
   187  		return appResult.Items, nil
   188  	}
   189  
   190  	appResult, err := service.DynamodbSvc.Query(&dynamodb.QueryInput{
   191  		TableName: &query.Workspace,
   192  		IndexName: aws.String("appNameCountryIndex"),
   193  		KeyConditions: map[string]*dynamodb.Condition{
   194  			"appName": {
   195  				ComparisonOperator: aws.String("EQ"),
   196  				AttributeValueList: []*dynamodb.AttributeValue{
   197  					{S: aws.String(query.AppName)},
   198  				},
   199  			},
   200  			"countryCode": {
   201  				ComparisonOperator: aws.String("EQ"),
   202  				AttributeValueList: []*dynamodb.AttributeValue{
   203  					{S: aws.String(query.Country)},
   204  				},
   205  			},
   206  		},
   207  		FilterExpression: aws.String("environment = :e"),
   208  		ExpressionAttributeValues: map[string]*dynamodb.AttributeValue{
   209  			":e": {
   210  				S: aws.String(query.Environment),
   211  			},
   212  		},
   213  	})
   214  	if err != nil {
   215  		getAPIKeySpan.AddField("error.message", fmt.Sprintf("Unable to getApiToken from discovery v3 admin: %s", err.Error()))
   216  		getAPIKeySpan.AddField("query.values", fmt.Sprintf("%s,%s,%s", query.AppName, query.Country, query.Environment))
   217  		return nil, err
   218  	}
   219  	return appResult.Items, nil
   220  }

   222  func parseAPIToken(ctx context.Context, result []map[string]*dynamodb.AttributeValue) (string, error) {
   223  	_, getQueryAPIKeySpan := beeline.StartSpan(ctx, "parseAPIToken")
   224  	defer getQueryAPIKeySpan.Send()
   225  	var discovery map[string]interface{}
   226  
   227  	if len(result) > 0 {
   228  		err := dynamodbattribute.UnmarshalMap(result[0], &discovery)
   229  		if err != nil {
   230  			getQueryAPIKeySpan.AddField("error.message", fmt.Sprintf("Unable to unmarshal results: %s", err.Error()))
   231  			return "", err
   232  		}
   233  		getQueryAPIKeySpan.AddField("success.message", "successfully retrieved ApiToken")
   234  		getQueryAPIKeySpan.Send()
   235  		return fmt.Sprintf("%v", discovery["apiToken"]), nil
   236  	}
   237  	return "", errors.New("No results found")
   238  }

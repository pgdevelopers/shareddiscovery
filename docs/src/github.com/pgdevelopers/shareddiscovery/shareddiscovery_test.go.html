   142  
   143  	// define a QueryInput
   144  	query := QueryInput{
   145  		Workspace:   "discovery_app",
   146  		Signature:   "e52af791ec66085081f993a42d2a02a4b1dc08ad7b9f030dac5a0c20d5a0c68c",
   147  		Brand:       "oralb",
   148  		Environment: "qa",
   149  		Country:     "US",

   152  			"environment": "qa",
   153  			"countryCode": "US",
   154  		},
   155  	}
   156  
   157  	// setup shareddiscovery now
   158  	shareddiscovery = New(dynamo)
   159

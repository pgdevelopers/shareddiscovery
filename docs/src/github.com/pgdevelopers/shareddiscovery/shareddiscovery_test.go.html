   281  	// setup AWS Session

   284  	// setup DynamoDB
   285  	dynamo := dynamodb.New(session)
   286  
   287  	// define a QueryInput
   288  	query := QueryInput{
   289  		Workspace:   "discovery_app",
   290  		Signature:   "e52af791ec66085081f993a42d2a02a4b1dc08ad7b9f030dac5a0c20d5a0c68c",
   291  		Brand:       "oralb",
   292  		Environment: "qa",
   293  		Country:     "US",
   294  		QueryString: map[string]string{
   295  			"brand":       "oralb",
   296  			"environment": "qa",
   297  			"countryCode": "US",
   298  		},
   299  	}
   300  
   301  	// setup shareddiscovery now
